### 4.2 集成测试示例

```go
func TestSQLiteObserverIntegration(t *testing.T) {
    // 创建临时目录
    tmpDir := t.TempDir()

    // 创建 SQLiteObserver
    observer, err := NewSQLiteObserver(tmpDir, nil, nil)
    require.NoError(t, err)
    defer observer.Close()

    ctx := context.WithValue(context.Background(), "session_key", "test-session")

    // 创建用户输入事件
    userEvent := &events.PromptSubmittedEvent{
        BaseEvent: events.BaseEvent{
            TraceID:   "trace-001",
            SpanID:    "span-001",
            Timestamp: time.Now(),
        },
        UserInput:  "Hello",
        SessionKey: "test-session",
    }

    err = observer.OnEvent(ctx, userEvent)
    require.NoError(t, err)

    // 验证数据已存储
    service := observer.GetConversationService()
    conversations, err := service.GetByTraceID(ctx, "trace-001")
    require.NoError(t, err)
    assert.Len(t, conversations, 1)
    assert.Equal(t, "user", conversations[0].Role)
    assert.Equal(t, "Hello", conversations[0].Content)
}
```
